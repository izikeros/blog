"""Module with T-SNE analysis and 2D/3D plots."""
from plotly import graph_objs as go


def tsne_fig_2d(X_embedded, indices, labels):

    # unwrap
    class_1_indices, class_2_indices = indices
    class_1_label, class_2_label = labels

    # prepare serie 1
    trace1 = go.Scatter(
        x=X_embedded[class_1_indices, 0],
        y=X_embedded[class_1_indices, 1],
        mode='markers',
        marker=dict(color='rgba(0,127,0, 1)',
                    size=10,
                    symbol='circle',
                    opacity=0.5),
        name=class_1_label,
    )

    # prepare serie 2
    trace2 = go.Scatter(
        x=X_embedded[class_2_indices, 0],
        y=X_embedded[class_2_indices, 1],
        mode='markers',
        marker=dict(color='rgba(127, 0, 0, 1)',
                    size=10,
                    symbol='circle',
                    opacity=0.5),
        name=class_2_label,
    )

    # prepare plot
    layout = dict(height=600, width=600)
    data = [trace1, trace2]
    fig = go.Figure(data=data, layout=layout)
    return fig


def tsne_fig_3d(X_embedded, class_1_indices, class_2_indices):
    trace1 = go.Scatter3d(
        x=X_embedded[class_1_indices, 0],
        y=X_embedded[class_1_indices, 1],
        z=X_embedded[class_1_indices, 2],
        mode='markers',
        marker=dict(color='rgb(0,127,0)', size=3, opacity=0.5),
    )
    trace2 = go.Scatter3d(
        x=X_embedded[class_2_indices, 0],
        y=X_embedded[class_2_indices, 1],
        z=X_embedded[class_2_indices, 2],
        mode='markers',
        marker=dict(color='rgba(127, 0, 0, 0.66)',
                    size=3,
                    symbol='circle',
                    opacity=0.5),
    )
    layout = dict(height=600, width=600)
    data = [trace1, trace2]
    fig = go.Figure(data=data, layout=layout)
    return fig
